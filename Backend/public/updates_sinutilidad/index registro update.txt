            const el = document.getElementById('toast');

            el.className = `show ${tipo}`;
            setTimeout(() => el.className = '', 2500);
        };
        
        const mostrarSeccion = (id) => {
            document.querySelectorAll('section').forEach(s => s.classList.add('hidden'));
            const el = document.getElementById(`sec-${id}`);
            if (el) el.classList.remove('hidden');
        };
        
        // ========== REGISTRO Y LOGIN ==========
        async function cargarPaises() {
            try {
                const res = await fetch(`${API}/paises`);
                const paises = await res.json();
                const select = document.getElementById('reg-pais');
                select.innerHTML = '<option value="">Selecciona un país</option>' + paises.map(p => `<option value="${p.paisid}">${p.nombre}</option>`).join('');
            } catch (err) {
                toast('Error al cargar países', 'error');
            }
        }
        
        async function registrarCliente() {
            const nombre = document.getElementById('reg-nombre').value.trim();
            const apellido = document.getElementById('reg-apellido').value.trim();
            const email = document.getElementById('reg-email').value.trim();
            const telefono = document.getElementById('reg-telefono').value.trim();
            const paisid = document.getElementById('reg-pais').value;
            
            if (!nombre || !apellido || !email || !paisid) {
                toast('Completa todos los campos obligatorios', 'error');
                return;
            }
            
            try {
                const res = await fetch(`${API}/clientes`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ nombre, medio: null, apellido, correo: email, paisid, numero_contacto: telefono || null })
                });
                
                if (!res.ok) throw new Error(await res.text());
                
                toast('Cliente registrado con éxito. Ahora inicia sesión.');
                mostrarLogin();
            } catch (err) {
                toast('Error al registrar: ' + err.message, 'error');
            }
        }
        
        function mostrarRegistro() {
            document.getElementById('sec-login').classList.add('hidden');
            document.getElementById('sec-registro').classList.remove('hidden');
            cargarPaises();
        }
        
        function mostrarLogin() {
            document.getElementById('sec-registro').classList.add('hidden');
            document.getElementById('sec-login').classList.remove('hidden');
        }
        
        async function loginPorEmail() {
            const email = document.getElementById('login-email').value.trim();
            if (!email) return toast('Ingresa tu correo', 'error');
            
            try {
                const res = await fetch(`${API}/login`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ email })
                });
                
                if (!res.ok) throw new Error('Cliente no encontrado');
                const data = await res.json();
                
                usuarioActual = { id: data.Id_cliente, rol: data.rol, nombre: data.nombre, apellido: data.apellido, email: data.email };
                actualizarNav();
                
                if (usuarioActual.rol === 'admin') mostrarSeccion('admin');
                else {
                    mostrarSeccion('cliente');
                    await Promise.all([cargarCatalogo(), cargarPagosCliente(), cargarCarrito(), cargarPromociones()]);
                    cargarPerfil();
                }
            } catch (err) {
                toast('Error: ' + err.message, 'error');
            }
        }
        
        function cerrarSesion() { usuarioActual = null; carritoActual = null; mostrarLogin(); document.getElementById('nav-links').innerHTML = ''; }
        
        function actualizarNav() {
            const links = document.getElementById('nav-links');
            if (!usuarioActual) return;
            if (usuarioActual.rol === 'admin') links.innerHTML = '<button class="nav-btn danger" onclick="cerrarSesion()">Cerrar sesión</button>';
            else links.innerHTML = `<button class="nav-btn danger" onclick="cerrarSesion()">Salir</button>`;
        }

        function cambiarPestana(pestana, btn) {
            document.getElementById('pestana-catalogo').classList.toggle('hidden', pestana !== 'catalogo');
            document.getElementById('pestana-historial').classList.toggle('hidden', pestana !== 'historial');
            document.getElementById('pestana-perfil').classList.toggle('hidden', pestana !== 'perfil');
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            if (btn) btn.classList.add('active');
            if (pestana === 'historial') cargarHistorial();
            if (pestana === 'perfil') { cargarPerfil(); cargarPagosCliente(); }
        }
        
        // ========== PERFIL Y MÉTODOS DE PAGO ==========

                <button class="btn btn-peligro" onclick="desactivarCuenta()" style="margin-top:1rem">Desactivar mi cuenta</button>
            `;
        }
        
        async function cargarPagosCliente() {
            try {
                const res = await fetch(`${API}/clientes/${usuarioActual.id}/pagos`);
                pagos = await res.json();
                const container = document.getElementById('perfil-pagos');
                if (!pagos.length) container.innerHTML = '<div class="text-center" style="color:var(--texto-sec)">No tienes tarjetas registradas</div>';
                else {
                    container.innerHTML = pagos.map((p, i) => `
                        <div class="pago-item">
                            <div><div class="pago-number">•••• •••• •••• ${String(p.numero_de_tarjeta ? p.numero_de_tarjeta.slice(-4) : i+1)}</div><div class="pago-method">${p.MetodoPago || 'Tarjeta'}</div></div>
                            <button class="btn btn-peligro" style="padding:0.2rem 0.5rem" onclick="prepararEliminarPago(${p.id_metodo}, '${p.numero_de_tarjeta}')">Eliminar</button>
                        </div>
                    `).join('');
                }
            } catch (err) { toast('Error al cargar pagos', 'error'); }
        }
        
        let pagoAEliminar = null;
        function prepararEliminarPago(id_metodo, numero_tarjeta) {
            pagoAEliminar = { id_metodo, numero_tarjeta };
            document.getElementById('modal-eliminar-pago').classList.remove('hidden');
            document.getElementById('btn-confirmar-eliminar').onclick = () => eliminarMetodoPago();
        }
        function cerrarModalEliminarPago() { pagoAEliminar = null; document.getElementById('modal-eliminar-pago').classList.add('hidden'); }
        
        async function eliminarMetodoPago() {
            if (!pagoAEliminar) return;
            try {
                const res = await fetch(`${API}/clientes/pagos`, {
                    method: 'DELETE',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ cliente_id: usuarioActual.id, id_metodo: pagoAEliminar.id_metodo, numero_tarjeta: pagoAEliminar.numero_tarjeta })
                });
                if (!res.ok) throw new Error();
                toast('Método de pago eliminado');
                cerrarModalEliminarPago();
                await cargarPagosCliente();
            } catch (err) { toast('Error al eliminar', 'error'); }
        }
        
        function mostrarModalAgregarPago() { document.getElementById('modal-agregar-pago').classList.remove('hidden'); }
        function cerrarModalAgregarPago() { document.getElementById('modal-agregar-pago').classList.add('hidden'); }
        
        async function agregarMetodoPago() {
            const id_metodo = document.getElementById('pago-metodo').value;
            const numero_tarjeta = document.getElementById('pago-numero').value.trim();
            const cvv = document.getElementById('pago-cvv').value.trim();
            const fecha_vencimiento = document.getElementById('pago-fecha').value.trim();
            if (!numero_tarjeta || !cvv || !fecha_vencimiento) return toast('Completa todos los campos', 'error');
            
            try {
                const res = await fetch(`${API}/pagos`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ cliente_id: usuarioActual.id, id_metodo, numero_tarjeta, cvv, fecha_vencimiento })
                });
                if (!res.ok) throw new Error();
                toast('Método de pago agregado');
                cerrarModalAgregarPago();
                await cargarPagosCliente();
                document.getElementById('pago-numero').value = '';
                document.getElementById('pago-cvv').value = '';
                document.getElementById('pago-fecha').value = '';
            } catch (err) { toast('Error al agregar', 'error'); }
        }
        
        async function desactivarCuenta() {
            if (!confirm('¿Estás seguro? Se cerrarán tus pedidos pendientes y desactivará tu cuenta.')) return;
            try {
                const res = await fetch(`${API}/clientes/desactivar/${usuarioActual.id}`, { method: 'PUT' });
                if (!res.ok) throw new Error();
                toast('Cuenta desactivada. Serás redirigido.');
                setTimeout(() => cerrarSesion(), 2000);
            } catch (err) { toast('Error al desactivar', 'error'); }
