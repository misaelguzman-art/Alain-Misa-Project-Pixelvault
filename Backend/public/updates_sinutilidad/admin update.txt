    <script>
        // ========== CONFIGURACIÓN ==========
        const API = 'http://localhost:3000/api';
        let usuarioActual = null, carritoActual = null, promociones = [], pagos = [];
        const EMOJIS = { juego: '🎮', dlc: '🔮', complemento: '🎁' };
        
        // Utilidades
        const toast = (msg, tipo = 'success') => {
            const el = document.getElementById('toast');
            el.textContent = msg;
            el.className = `show ${tipo}`;
            setTimeout(() => el.className = '', 2500);
        };
        
        const mostrarSeccion = (id) => {
            document.querySelectorAll('section').forEach(s => s.classList.add('hidden'));
            document.getElementById(`sec-${id}`)?.classList.remove('hidden');
        };
        
        // Navegación
        const actualizarNav = () => {
            const links = document.getElementById('nav-links');
            links.innerHTML = usuarioActual.rol === 'admin' 
                ? '<button class="nav-btn danger" onclick="cerrarSesion()">Cerrar sesión</button>'
                : `<button class="nav-btn" onclick="cambiarPestana('catalogo', this)">Catálogo</button>
                   <button class="nav-btn" onclick="cambiarPestana('historial', this); cargarHistorial()">Mis Pedidos</button>
                   <button class="nav-btn danger" onclick="cerrarSesion()">Salir</button>`;
        };
        
        // Login
        async function iniciarSesion() {
            const id = document.getElementById('login-id').value;
            const rol = document.getElementById('login-rol').value;
            if (!id) return toast('Ingresa un ID', 'error');
            usuarioActual = { id: parseInt(id), rol };
            actualizarNav();
            if (rol === 'admin') mostrarSeccion('admin');
            else {
                mostrarSeccion('cliente');
                await Promise.all([cargarCatalogo(), cargarPagos(), cargarCarrito(), cargarPromociones()]);
            }
        }
        
        function cerrarSesion() { usuarioActual = null; carritoActual = null; document.getElementById('nav-links').innerHTML = ''; mostrarSeccion('login'); }
        
        // Pestañas
        function cambiarPestana(pestana, btn) {
            document.getElementById('tab-catalogo').classList.toggle('hidden', pestana !== 'catalogo');
            document.getElementById('tab-historial').classList.toggle('hidden', pestana !== 'historial');
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            if (btn) btn.classList.add('active');
            if (pestana === 'historial') cargarHistorial();

        async function toggleEdiciones(e, idx, productId, nombre, tipo) {
            e.stopPropagation();
            const picker = document.getElementById(`picker-${idx}`);
            const editionsDiv = document.getElementById(`editions-${idx}`);
            if (picker.classList.contains('open')) return picker.classList.remove('open');
            document.querySelectorAll('.edition-picker').forEach(p => p.classList.remove('open'));
            picker.classList.add('open');
            if (editionsDiv.dataset.loaded) return;
            
            try {
                if (tipo !== 'juego') {
                    editionsDiv.innerHTML = `<div class="edition-opt" onclick="agregarCarrito(${productId}, '${nombre}', 'Estándar', 0)"><div><div class="ed-name">Contenido Digital</div><div class="ed-stock">Disponible</div></div></div>`;
                } else {
                    const res = await fetch(`${API}/productos/${productId}/ediciones`);
                    const eds = await res.json();
                    editionsDiv.innerHTML = eds.map(ed => `
                        <div class="edition-opt ${ed.stock === 0 ? 'no-stock' : ''}" 
                             onclick="${ed.stock === 0 ? '' : `agregarCarrito(${productId}, '${nombre.replace(/'/g,"\\'")}', '${ed.nombre_edicion}', ${ed.precio})`}">
                            <div><div class="ed-name">${ed.nombre_edicion}</div><div class="ed-stock">${ed.stock === 0 ? 'Sin stock' : `Stock: ${ed.stock}`}</div></div>
                            <div class="ed-price">$${parseFloat(ed.precio).toFixed(2)}</div>
                        </div>
                    `).join('');
                }
                editionsDiv.dataset.loaded = '1';
            } catch { editionsDiv.innerHTML = '<div class="text-center" style="color:var(--rojo)">Error</div>'; }
        }
        
        // Carrito
        async function cargarCarrito() {
            try {
                const res = await fetch(`${API}/clientes/${usuarioActual.id}/carrito`);
                const data = await res.json();
                if (data.carrito) carritoActual = { carro_id: data.carrito.carro_id, items: data.items };
                else {
                    const nuevo = await fetch(`${API}/clientes/${usuarioActual.id}/carrito`, { method: 'POST' }).then(r => r.json());
                    carritoActual = { carro_id: nuevo.carro_id, items: [] };
                }
                renderCarrito();
            } catch { toast('Error carrito', 'error'); }
        }
        
        async function agregarCarrito(productId, nombre, edicion, precio) {
            if (!carritoActual) return toast('Error: carrito no activo', 'error');
            try {
                await fetch(`${API}/carrito/${carritoActual.carro_id}/items`, {
                    method: 'POST', headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ productoid: productId, cantidad_pedida: 1, precio_prod: precio })
                });
                const existente = carritoActual.items.find(i => i.productoid === productId);
                if (existente) existente.cantidad_pedida++;
                else carritoActual.items.push({ productoid: productId, nombre_producto: nombre, nombre_edicion: edicion, precio_prod: precio, cantidad_pedida: 1 });
                renderCarrito();
                toast(`✅ ${nombre} agregado`);
                document.querySelectorAll('.edition-picker').forEach(p => p.classList.remove('open'));
            } catch { toast('Error al agregar', 'error'); }
        }
        
        async function eliminarItem(detallesId, productId) {
            try {
                await fetch(`${API}/carrito/${carritoActual.carro_id}/items/${detallesId}`, { method: 'DELETE' });
                carritoActual.items = carritoActual.items.filter(i => i.detallesid !== detallesId && i.productoid !== productId);
                renderCarrito();
                toast('Item eliminado');
            } catch { toast('Error al eliminar', 'error'); }
        }
        
        function renderCarrito() {
            const list = document.getElementById('cart-list');
            const footer = document.getElementById('cart-footer');
            const badge = document.getElementById('cart-count');
            if (!carritoActual?.items?.length) {
                list.innerHTML = '<div class="cart-empty">Tu carrito está vacío<br><span style="font-size:1.5rem">🛒</span></div>';
                footer.classList.add('hidden');
                badge.style.display = 'none';
                return;
            }
            const total = carritoActual.items.reduce((s, i) => s + i.precio_prod * i.cantidad_pedida, 0);
            list.innerHTML = carritoActual.items.map(i => `
                <div class="cart-item">
                    <div style="flex:1"><div class="cart-name">${i.nombre_producto}</div>${i.nombre_edicion ? `<div class="cart-edition">${i.nombre_edicion}</div>` : ''}<div class="cart-edition">x${i.cantidad_pedida}</div></div>
                    <div class="flex gap-1 center"><div class="cart-price">$${(i.precio_prod * i.cantidad_pedida).toFixed(2)}</div><button class="cart-remove" onclick="eliminarItem(${i.detallesid || null}, ${i.productoid})">✕</button></div>
                </div>
            `).join('');
            document.getElementById('cart-total').textContent = `$${total.toFixed(2)}`;
            footer.classList.remove('hidden');
            badge.style.display = 'inline';
            badge.textContent = carritoActual.items.reduce((s, i) => s + i.cantidad_pedida, 0);
        }

        // Checkout
        function abrirCheckout() {
            if (!carritoActual?.items?.length) return;
            const total = carritoActual.items.reduce((s, i) => s + i.precio_prod * i.cantidad_pedida, 0);
            const promoId = document.getElementById('promo-select').value;
            const promo = promociones.find(p => p.promocionid == promoId);
            const totalDesc = promo ? total * (1 - promo.descuento / 100) : total;
            const metodoOptions = pagos.map((p, i) => `<option value="${p.id_metodo || i+1}">${p.MetodoPago || 'Método'} #${i+1}</option>`).join('');
            
            const modal = document.getElementById('checkout-modal');
            modal.innerHTML = `
                <h3>Confirmar Pedido</h3>
                ${carritoActual.items.map(i => `<div class="order-row"><span>${i.nombre_producto} ${i.nombre_edicion ? `(${i.nombre_edicion})` : ''}</span><span>$${(i.precio_prod * i.cantidad_pedida).toFixed(2)}</span></div>`).join('')}
                <div class="divider"></div>
                ${promo ? `<div class="order-row"><span class="promo-tag">🏷 ${promo.nombre} (-${promo.descuento}%)</span><span class="promo-tag">-$${(total - totalDesc).toFixed(2)}</span></div>` : ''}
                <div class="order-row total"><span>Total a pagar</span><span>$${totalDesc.toFixed(2)}</span></div>
                <div class="divider"></div>
                <div class="mb-2"><label class="label">Método de pago</label><select id="checkout-metodo" class="select w-full">${metodoOptions}</select></div>
                <div class="flex gap-1"><button class="btn btn-secundario flex-1" onclick="cerrarCheckout()">Cancelar</button><button class="btn btn-exito flex-2" onclick="confirmarCompra(${promoId || 'null'})">Pagar $${totalDesc.toFixed(2)} →</button></div>
            `;
            document.getElementById('checkout-overlay').classList.remove('hidden');
        }
        
        function cerrarCheckout() { document.getElementById('checkout-overlay').classList.add('hidden'); }
        
        async function confirmarCompra(promoId) {
            const metodoId = parseInt(document.getElementById('checkout-metodo').value);
            if (!metodoId) return toast('Selecciona método de pago', 'error');
            const btn = document.querySelector('#checkout-modal .btn-exito');
            btn.disabled = true;
            btn.innerHTML = '<span class="spinner"></span>';
            try {
                const res = await fetch(`${API}/carrito/${carritoActual.carro_id}/confirmar`, {
                    method: 'PUT', headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ metodo_pago_id: metodoId })
                });
                const data = await res.json();
                if (!res.ok) throw new Error(data.error);
                if (promoId) await fetch(`${API}/pedidos/aplicar-promocion`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ pedido_id: data.pedido_id, promocion_id: promoId }) });
                
                document.getElementById('checkout-modal').innerHTML = `
                    <div class="success-screen"><div class="success-icon">✅</div><h3>¡Compra Exitosa!</h3><div class="order-number">Pedido #${data.pedido_id}</div><p>Tu pedido está siendo procesado.</p><p>Total cobrado: <strong>$${parseFloat(data.total).toFixed(2)}</strong></p><button class="btn btn-primario btn-full mt-2" onclick="reiniciarCarrito()">Seguir comprando</button></div>
                `;
                carritoActual.items = [];
                renderCarrito();
            } catch (err) { toast(`Error: ${err.message}`, 'error'); btn.disabled = false; btn.textContent = 'Reintentar'; }
        }
        
        async function reiniciarCarrito() {
            cerrarCheckout();
            try {
                const res = await fetch(`${API}/clientes/${usuarioActual.id}/carrito`, { method: 'POST' });
                const data = await res.json();
                carritoActual = { carro_id: data.carro_id, items: [] };
                renderCarrito();
                await cargarPagos();
            } catch {}
        }

            try {
                const res = await fetch(`${API}/${endpoint}`);

        // Función mejorada para cargar cualquier reporte
        async function cargarReporte(endpoint) {
            const tablaHead = document.getElementById('report-head');
            const tablaBody = document.getElementById('report-body');
            
            tablaBody.innerHTML = '<tr><td colspan="100%" style="padding:2rem;text-align:center"><div class="spinner"></div> Cargando...</td></tr>';
            
            try {
                const res = await fetch(`${API}/${endpoint}`);
                if (!res.ok) throw new Error(`HTTP ${res.status}`);
                const data = await res.json();
                
                if (!data || data.length === 0) {
                    tablaBody.innerHTML = '<tr><td colspan="100%" style="padding:1rem;text-align:center;color:var(--texto-sec)">Sin datos disponibles</td></tr>';
                    tablaHead.innerHTML = '';
                    return;
                }
                
                const columnas = Object.keys(data[0]);
                tablaHead.innerHTML = `<tr>${columnas.map(col => `<th>${col}</th>`).join('')}</tr>`;
                tablaBody.innerHTML = data.map(fila => `
                    <tr>${columnas.map(col => `<td>${fila[col] ?? '—'}</td>`).join('')}</tr>
                `).join('');
                
            } catch (err) {
                console.error(`Error en ${endpoint}:`, err);
                tablaBody.innerHTML = `<tr><td colspan="100%" style="padding:1rem;text-align:center;color:var(--rojo)">Error al cargar: ${err.message}</td></tr>`;
                toast(`Error al cargar ${endpoint}`, 'error');
            }
        }
